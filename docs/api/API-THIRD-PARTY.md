## 1. Socket.io (v4)

**Docs:** https://socket.io/docs/v4/

### Server (NestJS)
```ts
// Install
npm install @nestjs/websockets @nestjs/platform-socket.io socket.io

// Gateway
import { WebSocketGateway, SubscribeMessage, MessageBody, ConnectedSocket } from '@nestjs/websockets'
import { Socket } from 'socket.io'

@WebSocketGateway({ cors: { origin: '*' } })
export class InterviewGateway {
  @SubscribeMessage('session:start')
  handleStart(@MessageBody() data: StartDto, @ConnectedSocket() client: Socket) {
    client.emit('session:created', { sessionId: '...' })
  }
}
```

### Client (Next.js)
```ts
// Install
npm install socket.io-client

import { io, Socket } from 'socket.io-client'
const socket: Socket = io(process.env.NEXT_PUBLIC_API_URL!, { transports: ['websocket'] })
socket.on('connect', () => console.log('connected', socket.id))
socket.emit('session:start', { candidateName: 'Jane', jobRole: 'backend-engineer' })
socket.on('agent:response', (data) => { /* handle */ })
```

## 2. Prisma ORM (v5)

**Docs:** https://www.prisma.io/docs

```ts
// Install
npm install prisma @prisma/client
npx prisma init  // creates prisma/schema.prisma + .env

// Generate client after schema changes
npx prisma generate

// Run migrations
npx prisma migrate dev --name init

// Prisma Client usage
import { PrismaClient } from '@prisma/client'
const prisma = new PrismaClient()

// Create session
const session = await prisma.session.create({ data: { candidateName, jobRole, status: 'pending' } })

// Create turn
await prisma.turn.create({ data: { sessionId: session.id, turnIndex: 0, questionText, candidateAnswer, score, feedback } })

// Query with filter
const sessions = await prisma.session.findMany({
  where: { jobRole: 'backend-engineer', status: 'completed' },
  include: { turns: true },
  orderBy: { startedAt: 'desc' }
})
```

## 3. Faster-Whisper (Python)

**Docs:** https://github.com/SYSTRAN/faster-whisper

```python
# Install
pip install faster-whisper

from faster_whisper import WhisperModel

model = WhisperModel("base.en", device="cuda", compute_type="float16")  # GPU
# model = WhisperModel("base.en", device="cpu", compute_type="int8")   # CPU fallback

def transcribe(audio_path: str) -> str:
    segments, info = model.transcribe(audio_path, beam_size=5, language="en")
    return " ".join(s.text for s in segments).strip()
```

**Models by size/speed:**
| Model | Size | Speed (GPU) | Accuracy |
|---|---|---|---|
| `tiny.en` | 75MB | Fastest | Lower |
| `base.en` | 145MB | Fast | Good |
| `small.en` | 483MB | Medium | Better |
| `medium.en` | 1.5GB | Slower | Best (recommended) |

## 4. Piper TTS (Python)

**Docs:** https://github.com/rhasspy/piper
**Voices:** https://huggingface.co/rhasspy/piper-voices

```python
# Install
pip install piper-tts

import wave
from piper.voice import PiperVoice

voice = PiperVoice.load("en_US-amy-medium.onnx", config_path="en_US-amy-medium.onnx.json")

def synthesize(text: str, output_path: str):
    with wave.open(output_path, "w") as wav_file:
        voice.synthesize(text, wav_file)
```

**Getting voices:**
```bash
# Download voice model + config
wget https://huggingface.co/rhasspy/piper-voices/resolve/main/en/en_US/amy/medium/en_US-amy-medium.onnx
wget https://huggingface.co/rhasspy/piper-voices/resolve/main/en/en_US/amy/medium/en_US-amy-medium.onnx.json
```

> Piper does not natively output viseme timestamps. Viseme data is generated by mapping phoneme durations from Piper's internal synthesis steps or via `piper --output_json` flag (outputs word timing). For MVP, use pre-computed phoneme-to-viseme mapping on the text before synthesis.


**Morph target (lip-sync):**
```ts
import { useFrame } from '@react-three/fiber'
import { useRef } from 'react'
import * as THREE from 'three'

function LipSync({ visemes, audioStartTime }) {
  const meshRef = useRef<THREE.SkinnedMesh>()

  useFrame(({ clock }) => {
    const elapsed = clock.getElapsedTime() - audioStartTime
    const current = visemes.findLast(v => v.time <= elapsed)
    if (current && meshRef.current) {
      const idx = meshRef.current.morphTargetDictionary[current.value]
      meshRef.current.morphTargetInfluences[idx] = 1
    }
  })
  return null
}
```

## 6. MediaRecorder API (Browser — Audio Capture)

**MDN Docs:** https://developer.mozilla.org/en-US/docs/Web/API/MediaRecorder

```ts
// useAudioRecorder.ts
export function useAudioRecorder() {
  const mediaRecorderRef = useRef<MediaRecorder | null>(null)
  const chunksRef = useRef<Blob[]>([])

  const start = async () => {
    const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
    const recorder = new MediaRecorder(stream, { mimeType: 'audio/webm' })
    recorder.ondataavailable = (e) => chunksRef.current.push(e.data)
    recorder.start()
    mediaRecorderRef.current = recorder
  }

  const stop = (): Promise<Blob> => new Promise((resolve) => {
    const recorder = mediaRecorderRef.current!
    recorder.onstop = () => resolve(new Blob(chunksRef.current, { type: 'audio/webm' }))
    recorder.stop()
    chunksRef.current = []
  })

  return { start, stop }
}
```

## 7. FastAPI (Python Voice Service)

**Docs:** https://fastapi.tiangolo.com

```python
# Install
pip install fastapi uvicorn python-multipart

from fastapi import FastAPI, UploadFile, File
from pydantic import BaseModel
import uvicorn

app = FastAPI()

class SynthesizeRequest(BaseModel):
    text: str
    voice: str = "en_US-amy-medium"

@app.post("/transcribe")
async def transcribe(audio: UploadFile = File(...)):
    # save to temp file, run whisper, return text
    return { "text": "...", "duration_ms": 1200 }

@app.post("/synthesize")
async def synthesize(req: SynthesizeRequest):
    # run piper, return audio url + visemes
    return { "audioUrl": "/audio/abc.wav", "visemes": [] }

@app.get("/health")
def health():
    return { "status": "ok" }

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
```
